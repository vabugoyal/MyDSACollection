NOTE:
    Kernel mutexes are hybrid: they use atomic instructions in
    userspace for fast locking and fall back to kernel-managed
    sleep queues when contention occurs.

    The atomic compare-and-swap instruction is provided by the hardware (CPU), not the OS.
    The OS uses that hardware instruction to build mutexes.
    C++ std::mutex
         ↓
    OS mutex / futex
         ↓
    Hardware atomic instructions (CAS)

    - the mutex state lives in the user space
    - the kernel never owns the mutex
    - the kernel only puts threads to sleep and wakes
      them up when the userspace asks (using futex)
    a futex is a sleep until memory value changes mechanism
    Thread A                 Kernel                  Thread B
    -----------------------------------------------------------
    CAS(lock,0→1) success
    work
    unlock → lock=0
    futex(WAKE)

                             wake Thread B
                                                     CAS(lock,0→1) success
                                                     work

        #include <mutex>
        #include <thread>
        #include <condition_variable>

        using namespace std;

        mutex m;
        std::condition_variable cv;
        int balance = 0;

        void addMoney(int money)
        {
            lock_guard<mutex> l(m);
            balance += money;
            cout << "Amount added, current balance: " << money << endl;
            cv.notify_one();
        }

        void withdrawMoney(int money) {
            std::unique_lock<mutex> ul(m);
            cv.wait(ul, []{return balance != 0;});
            if (balance >= money) {
                balance -= money;
                cout << "Amount Deducted: " << money << endl;
            } else {
                cout << "Amount can't be deducted, current balance is less than" << money << endl;
            }
        }

        int main() {
            thread t1(withdrawMoney, 500);
            thread t2(addMoney, 500);

            t1.join();
            t2.join();
        }

            #include <iostream>
            #include <memory>

            // ----------------------
            // Type-erased polymorphic base
            template<typename R, typename... Args>
            struct CallableBase {
                virtual R call(Args... args) = 0;                       // call the stored callable
                virtual std::unique_ptr<CallableBase> clone() const = 0; // copy safely
                virtual ~CallableBase() = default;
            };

            // Wrapper for any specific callable (lambda, function pointer, functor)
            template<typename R, typename... Args, typename Callable>
            struct CallableDerived : CallableBase<R, Args...> {
                Callable c;
                CallableDerived(Callable c_) : c(c_) {}

                R call(Args... args) override { return c(args...); }

                std::unique_ptr<CallableBase<R, Args...>> clone() const override {
                    return std::make_unique<CallableDerived>(*this);
                }
            };

            // ----------------------
            // Simplified template version of std::function
            template<typename Signature>
            class Function; // primary template

            template<typename R, typename... Args>
            class Function<R(Args...)> {
                std::unique_ptr<CallableBase<R, Args...>> target; // holds any callable

                public:
                    Function() = default;

                    // Construct from any callable
                    template<typename Callable>
                    Function(Callable c)
                        : target(std::make_unique<CallableDerived<R, Args..., Callable>>(c)) {}

                    // Copy constructor
                    Function(const Function& other)
                        : target(other.target ? other.target->clone() : nullptr) {}

                    // Assignment
                    Function& operator=(const Function& other) {
                        if (this != &other)
                            target = other.target ? other.target->clone() : nullptr;
                        return *this;
                    }

                    // Call operator
                    R operator()(Args... args) {
                        if(target) return target->call(args...);
                        throw std::bad_function_call();
                    }
            };

            // ----------------------
            int main() {
                // Function storing a lambda
                Function<void(int,int)> f = [](int x,int y){
                    std::cout << "Sum = " << x+y << "\n";
                };

                f(3,5);

                // Copy function
                Function<void(int,int)> g = f;
                g(10,20);

                // Store a functor
                struct Multiply {
                    void operator()(int a,int b){ std::cout << "Mul = " << a*b << "\n"; }
                };
                Function<void(int,int)> h = Multiply{};
                h(4,5);
            }

Thread Pooling
    A pool of threads and a queue of tasks
        struct ThreadPool {
            bool stop;
            deque<std::function<void()>> tasks;
            mutex m;
            vector<thread> threads;
            condition_variable cv;
            int numThreads;

            void processTask() {
                while (true) {
                    std::function<void()> task;
                    {
                        unique_lock<mutex> lk(m);
                        cv.wait(lk, [this](){ return stop || !tasks.empty(); });

                        // exit only if stop is true AND all tasks are processed
                        if (stop && tasks.empty())
                            break;

                        if (!tasks.empty()) {
                            task = std::move(tasks.front());
                            tasks.pop_front();
                        } else {
                            continue;
                        }
                    }
                    cout << "Thread ID: " << std::this_thread::get_id() << " ";
                    task();
                }
            }

            ThreadPool(int numthreads) : numThreads(numthreads), stop(false) {
                for (int i = 0; i < numThreads; i++) {
                    threads.emplace_back(&ThreadPool::processTask, this);
                }
            }

            template<typename F>
            void enqueueTask(F&& task) {
                {
                    std::unique_lock<std::mutex> l(m);
                    tasks.emplace_back(std::forward<F>(task));  // perfect forwarding
                }
                cv.notify_one();
            }

            void terminate() {
                {
                    lock_guard<mutex> lk(m);
                    stop = true;
                }
                cv.notify_all();
                for (auto &thread : threads) {
                    thread.join();
                }
            }

            bool allComplete() {
                lock_guard<mutex> lk(m);
                return tasks.empty();
            }

            ~ThreadPool() {
                terminate(); // ensures proper cleanup
            }
        };

        int main() {
            int numTasks = 10;
            int numThreads = 2;
            ThreadPool tp(numThreads);

            for (int i = 0; i < numTasks; i++) {
                tp.enqueueTask(
                        [=]() {
                            cout << "Task " << i << " completed." << endl;
                            std::this_thread::sleep_for(std::chrono::seconds(1));
                        }
                );
            }

            cout << "All tasks completed!" << endl;
        }

    Lock Free Queue

    Interviewer might ask:
        Why do you need a dummy node?
            A dummy node acts as a sentinel to simplify the logic.
            It ensures head always points before the first real element,
            so enqueue and dequeue operations don’t have to handle special
            cases for empty or single-element queues. It also reduces
            race conditions and makes the lock-free algorithm cleaner and safer.



        #include <atomic>
        #include <memory>
        #include <iostream>

        template<typename T>
        struct Node {
            T data;
            std::atomic<Node*> next;

            Node() : next(nullptr) {}           // dummy node
            Node(T val) : data(val), next(nullptr) {}
        };

        template<typename T>
        class LockFreeQueue {
            std::atomic<Node<T>*> head;
            std::atomic<Node<T>*> tail;

        public:
            LockFreeQueue() {
                Node<T>* dummy = new Node<T>();  // initial dummy node
                head.store(dummy, std::memory_order_relaxed);
                tail.store(dummy, std::memory_order_relaxed);
            }

            ~LockFreeQueue() {
                // Delete remaining nodes safely
                while (Node<T>* node = head.load()) {
                    head.store(node->next);
                    delete node;
                }
            }

            void enqueue(T value) {
                Node<T>* newNode = new Node<T>(value);

                while (true) {
                    Node<T>* tailSnapshot = tail.load(std::memory_order_acquire);
                    Node<T>* nextSnapshot = tailSnapshot->next.load(std::memory_order_acquire);

                    // Validate snapshot: tail has not changed
                    if (tailSnapshot == tail.load(std::memory_order_acquire)) {

                        if (nextSnapshot == nullptr) {
                            // tailSnapshot is indeed the last node
                            // CAS inserts newNode at the end
                            if (tailSnapshot->next.compare_exchange_strong(
                                    nextSnapshot,
                                    newNode,
                                    std::memory_order_release,
                                    std::memory_order_acquire))
                            {
                                // Try to advance tail; okay if fails (another thread advanced it)
                                tail.compare_exchange_strong(
                                    tailSnapshot,
                                    newNode,
                                    std::memory_order_release,
                                    std::memory_order_acquire
                                );
                                return;
                            }
                        } else {
                            // Tail is lagging behind, help it move forward
                            // or it might be the case the our snapshot became stale
                            tail.compare_exchange_strong(
                                tailSnapshot,
                                nextSnapshot,
                                std::memory_order_release,
                                std::memory_order_acquire
                            );
                        }
                    }
                }
            }

            bool dequeue(T& result) {
                while (true) {
                    Node<T>* headSnapshot = head.load(std::memory_order_acquire);
                    Node<T>* firstSnapshot = headSnapshot->next.load(std::memory_order_acquire);

                    // Validate snapshot: head has not changed
                    if (headSnapshot == head.load(std::memory_order_acquire)) {

                        // If no node exists after head, queue is empty
                        // No need to reload head->next because if it was nullptr here, it will remain nullptr
                        // until some enqueue completes, which will be visible in the next iteration
                        if (firstSnapshot == nullptr) return false;

                        // Attempt to move head forward
                        if (head.compare_exchange_weak(
                                headSnapshot,
                                firstSnapshot,
                                std::memory_order_acq_rel,   // release old head + acquire new head
                                std::memory_order_acquire))
                        {
                            // Safe to read firstSnapshot->data because we acquired it
                            result = firstSnapshot->data;

                            // Delete old dummy / old head
                            delete headSnapshot;
                            return true;
                        }
                    }
                }
            }
        };


    Lock Free Stack

        #include <atomic>
        #include <memory>
        #include <iostream>

        template<typename T>
        struct Node {
            T data;
            std::atomic<Node*> next;

            Node() : next(nullptr) {}           // dummy node
            Node(T val) : data(val), next(nullptr) {}
        };

        template<typename T>
        class LockFreeStack {
            std::atomic<Node<T>*> head;

        public:
            LockFreeStack() {
                head.store(nullptr, std::memory_order_relaxed);
            }

            ~LockFreeStack() {
                // Delete remaining nodes safely
                while (Node<T>* node = head.load()) {
                    head.store(node->next);
                    delete node;
                }
            }

            void push(T value) {
                Node<T>* newNode = new Node<T>(value);
                while(true) {
                    Node<T> * headSnapshot = head.load(memory_order_acquire);
                    newNode->next = headSnapshot;

                    // CAS: set head to newNode if it still equals headSnapshot
                    if (head.compare_exchange_strong(headSnapshot,
                                                     newNode,
                                                     memory_order_release,
                                                     memory_order_acquire)) {
                        return;
                    }
                    // otherwise retry
                }
            }

            bool pop(T& result) {
                while (true) {
                    Node<T>* headSnapshot = head.load(std::memory_order_acquire);
                    Node<T>* nextSnapshot = headSnapshot->next.load(std::memory_order_acquire);

                    if (headSnapshot == nullptr) return false;

                    if (head.compare_exchange_strong(headSnapshot,
                                                     nextSnapshot,
                                                     memory_order_release,
                                                     memory_order_acquire)) {
                        result->data = headSnapshot;
                        delete headSnapshot;
                        return true;
                    }
            }
        };
